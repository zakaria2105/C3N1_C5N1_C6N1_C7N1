CREATE TABLE IF NOT EXISTS developpeurs ( 
	id INT(11) NOT NULL AUTO_INCREMENT , 
	nom VARCHAR(255) NOT NULL , 
	prenom VARCHAR(255) NOT NULL , 
	email VARCHAR(255) NOT NULL UNIQUE, 
	password VARCHAR(255) NOT NULL , 
	PRIMARY KEY (id)
	) ENGINE = InnoDB; 

CREATE TABLE IF NOT EXISTS techno (
	developpeur_id INT(11) NOT NULL , 
	HTML INT(11) NOT NULL , 
	JS INT(11) NOT NULL , 
	AJAX INT(11) NOT NULL , 
	PHP INT(11) NOT NULL , 
	CGI INT(11) NOT NULL ,
	FOREIGN KEY (developpeur_id) REFERENCES developpeurs(id) ON DELETE CASCADE ON UPDATE CASCADE
	) ENGINE = InnoDB; 
    
    
CREATE TABLE IF NOT EXISTS admin ( 
	id INT(11) NOT NULL AUTO_INCREMENT , 
	nom VARCHAR(255) NOT NULL , 
	prenom VARCHAR(255) NOT NULL , 
	email VARCHAR(255) NOT NULL UNIQUE, 
	password VARCHAR(255) NOT NULL , 
	PRIMARY KEY (id)
	) ENGINE = InnoDB; 

CREATE TABLE IF NOT EXISTS formation ( 
	developpeur_id INT(11) NOT NULL ,
	techno VARCHAR(255) NOT NULL , 
	date DATE NOT NULL , 
    	FOREIGN KEY (developpeur_id) REFERENCES developpeurs(id) ON DELETE CASCADE ON UPDATE CASCADE
	) ENGINE = InnoDB;
